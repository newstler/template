```ruby
module DashboardHelper
  # Renders a KPI card via the shared partial.
  def kpi_card(label:, value:, trend: nil, icon: nil, href: nil)
    render "shared/kpi_card", label: label, value: value, trend: trend, icon: icon, href: href
  end

  # Returns the percent change between current and previous values.
  # Returns nil if previous is zero or nil.
  def pct_change(current, previous)
    return nil if previous.nil? || previous.zero?
    ((current - previous) / previous.to_f * 100).round(1)
  end

  # Returns an arrow emoji for a change delta.
  def trend_arrow(delta)
    return "" if delta.nil?
    delta > 0 ? "↑" : (delta < 0 ? "↓" : "→")
  end

  # Renders a progress ring SVG. Generic — originally crew_progress_ring
  # and adventure_progress_ring in sailing_plus.
  def progress_ring(value:, max:, size: 48, label: nil, color: "accent")
    percent = max.to_i.zero? ? 0 : [(value.to_f / max * 100).round, 100].min
    radius = (size - 8) / 2
    circumference = 2 * Math::PI * radius
    offset = circumference - (percent / 100.0) * circumference

    render "shared/progress_ring",
      size: size,
      radius: radius,
      circumference: circumference,
      offset: offset,
      percent: percent,
      label: label,
      color: color
  end

  # Renders an SVG sparkline for a series of numbers.
  # Originally views_sparkline in sailing_plus.
  def sparkline(series, width: 120, height: 32, color: "accent")
    return "".html_safe if series.blank?
    max = series.max.to_f
    min = series.min.to_f
    range = (max - min).zero? ? 1 : (max - min)

    points = series.each_with_index.map do |value, i|
      x = (i.to_f / (series.size - 1)) * width
      y = height - ((value - min) / range) * height
      "#{x.round(1)},#{y.round(1)}"
    end.join(" ")

    svg = <<~SVG
      <svg xmlns="http://www.w3.org/2000/svg" width="#{width}" height="#{height}"
           data-controller="sparkline" class="sparkline">
        <polyline points="#{points}" fill="none" stroke="currentColor" stroke-width="1.5" />
      </svg>
    SVG
    svg.html_safe
  end

  # Renders the attention items strip from a list of hashes.
  def attention_items_strip(items)
    return "".html_safe if items.blank?
    render "shared/attention_items_strip", items: items
  end

  # Wraps expensive dashboard aggregations in a cache.
  # Key includes team_id + key_name + @range so invalidation is automatic.
  def cached_dashboard(key, expires_in: 5.minutes, &block)
    team_id = current_team&.id
    range_key = @range&.then { |r| r.begin.to_date.to_s }
    cache_key = ["dashboard", team_id, key, range_key].compact
    Rails.cache.fetch(cache_key, expires_in: expires_in, &block)
  end

  # Parses a range query param ("7d", "30d", "90d", "custom") into a Range.
  def time_range_from(param)
    case param.to_s
    when "7d"  then 7.days.ago..Time.current
    when "90d" then 90.days.ago..Time.current
    when "30d", "" then 30.days.ago..Time.current
    else 30.days.ago..Time.current
    end
  end
end
```

import { Controller } from "@hotwired/stimulus"

// Posts back a ?range= query param on select change, then reloads the page
// so the dashboard controller picks up the new @range.
//
// Usage:
//   <select data-controller="time-range"
//           data-action="change->time-range#update">
//     <option value="7d">Last 7 days</option>
//     <option value="30d" selected>Last 30 days</option>
//     <option value="90d">Last 90 days</option>
//   </select>
export default class extends Controller {
  update(event) {
    const range = event.target.value
    const url = new URL(window.location.href)
    url.searchParams.set("range", range)
    Turbo.visit(url.toString())
  }
}
```

class HomeController < ApplicationController
  include DashboardHelper

  before_action :authenticate_user!
  before_action :set_team

  def index
    @range = time_range_from(params[:range] || "30d")

    @total_users = @team.users.count
    @total_chats = @team.chats.count
    @recent_users = cached_dashboard(:recent_users) do
      @team.users.where(created_at: @range).count
    end
    @recent_chats = cached_dashboard(:recent_chats) do
      @team.chats.where(created_at: @range).count
    end

    @users_timeline = cached_dashboard(:users_timeline) do
      @team.users.where(created_at: @range).group_by_day(:created_at).count
    end

    @attention_items = build_attention_items
  end

  private

  def set_team
    @team = Team.find_by!(slug: params[:team_slug])
    raise ActiveRecord::RecordNotFound unless current_user.member_of?(@team)
  end

  def build_attention_items
    items = []
    unless @team.verified_at.present?
      items << { severity: :warning, label: t(".attention.verify_team"), path: edit_team_settings_path(@team.slug) }
    end
    items
  end
end

## Task 10: Performance rule for dashboards

- [ ] **Step 1: Append to performance rule**

Append to `.claude/rules/performance.md`:

```markdown
## Dashboards

Dashboards fetch and compute aggregate data that's accessed frequently. Apply these rules strictly:

1. **Always use `includes`** for associations accessed in the view.
2. **Use database aggregates** for "top N" queries, not Ruby iteration on preloaded data.
3. **Wrap expensive aggregations in `cached_dashboard`** — the helper's cache key includes `team_id` and `@range` so invalidation is automatic.
4. **Every dashboard controller test** must assert the query count is bounded:
   ```ruby
   test "dashboard runs N queries" do
     sign_in users(:one)
     assert_queries(15) { get team_root_path(teams(:one).slug) }
   end
   ```

Counter caches, collection rendering, and `.find { }` over preloaded data (per the existing performance rules) all apply.
```

## Dashboards

Chartkick + Groupdate with a shared helper module and reusable partials.

### Basic pattern

```erb
<%= kpi_card label: t(".users"), value: @user_count, trend: pct_change(@recent, @prev) %>

<%= render "shared/chart_card", title: t(".growth") do %>
  <%= line_chart @users_timeline %>
<% end %>

<%= attention_items_strip(@attention_items) %>
```

### Time-range selector

```erb
<select data-controller="time-range" data-action="change->time-range#update">
  <option value="7d">Last 7 days</option>
  <option value="30d" selected>Last 30 days</option>
  <option value="90d">Last 90 days</option>
</select>
```

In the controller: `@range = time_range_from(params[:range] || "30d")` → use `@range` in `.where(created_at: @range)` queries and `.group_by_day`.

### Caching

```ruby
@top_users = cached_dashboard(:top_users, expires_in: 10.minutes) do
  @team.users.joins(:chats).group("users.id").order("SUM(chats.total_cost) DESC").limit(10).to_a
end
```

Cache key includes team id + key + range, so invalidation is automatic on range change.

### Partials

- `shared/kpi_card` — label, value, optional trend, optional icon, optional link
- `shared/chart_card` — wraps a Chartkick chart with title + chart-theme Stimulus controller
- `shared/attention_items_strip` — color-coded action badges
- `shared/progress_ring` — SVG circular progress indicator

### Helpers

- `kpi_card(label:, value:, trend:, icon:, href:)`
- `pct_change(current, previous)`
- `trend_arrow(delta)` — returns ↑ ↓ →
- `sparkline(series, width:, height:)` — tiny SVG line chart
- `progress_ring(value:, max:, size:, label:)`
- `cached_dashboard(key, expires_in:, &block)`
- `time_range_from(param)` — returns a Range

### Rules (enforced)

See `.claude/rules/performance.md` § Dashboards.
